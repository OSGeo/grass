file(GLOB SRCMAKES "*.make")
set(include_make_depends)
foreach(srcmake ${SRCMAKES})
  get_filename_component(srcmake_DIR ${srcmake} DIRECTORY)
  get_filename_component(srcmake_NAME ${srcmake} NAME)
  string(REPLACE "${CMAKE_CURRENT_SOURCE_DIR}/" "" dstmake_DIR "${srcmake_DIR}")
  set(output_dir ${OUTDIR}/${GRASS_INSTALL_INCLUDEDIR}/${dstmake_DIR})
  add_custom_command(
    OUTPUT ${output_dir}/${srcmake_NAME}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${output_dir}
    COMMAND ${CMAKE_COMMAND} -E copy ${srcmake} ${output_dir}
    COMMENT "Copy ${srcmake} to ${output_dir}/${srcmake_NAME}")
  list(APPEND include_make_depends ${output_dir}/${srcmake_NAME})
endforeach()

add_custom_target(copy_make DEPENDS ${include_make_depends} LIB_PYTHON)
